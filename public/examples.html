Quick-start snippets

Hello world

Minimal entry point with std library and io helper.

.needs <std>
import * from "io" under io;

fn main() -> int {
    io.print("Hello, world!\n");
    return 0;
};

Personalized greeting

Prompt the user and use string interpolation.

.needs <std>
import string from "String";
import { readString, print } from "String" under str;

fn main() -> int {
    let name = str.readString("Name? ");
    str.print(`Hello, {name}!\n`);
    return 0;
};

Guarded access

Compare input using if/else.

if name == "AFlat" {
    str.print("Welcome back!\n");
} else {
    str.print("Access denied.\n");
};

Guessing game loop

for mutable int i = 0; i < 5; i = i + 1 {
    let guess = str.readString("Guess: ");
    let guessInt = guess.toInt()!;

    if guessInt == secret {
        str.print("You got it!\n");
        break;
    } else if guessInt < secret
        str.print("Too low\n")
    else
        str.print("Too high\n");
};

Student class

class Student {
    mutable int id;
    mutable string name = "";
    mutable int average;

    fn init(int id, string name, int average) -> Student {
        my.id = id;
        my.name = name;
        my.average = average;
    };
};

Roster printer

let roster = [
    new Student(1, "John", 90),
    new Student(2, "Jane", 95)
];

foreach s in roster {
    s.printInfo();
};

Ownership + result mashup

import {accept, reject} from "Utils/result" under pr;

fn addAssignment(Student &&student, Assignment &&assignment) -> bool! {
    student.addAssignment($assignment);
    return true;
};

fn divide(int a, int b) -> int! {
    if b == 0
        return pr.reject::(Error("Cannot divide by zero"));
    return a / b;
};

These helpers showcase how ownership moves through && parameters while the ! shorthand lets you bubble errors without extra boilerplate.
