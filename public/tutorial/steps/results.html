Error handling with result

import {accept, reject} from "Utils/result" under pr;

fn divide(int a, int b) -> int! {
    if b == 0
        return pr.reject::(Error("Cannot divide by zero"));
    return a / b;
};

fn main() -> int {
    match divide(10, 0) {
        Ok(value) => str.print(`Value: {value}\n`),
        Err(e) => str.print(`Error: {e}\n`)
    };
    return 0;
};

Apply the ! shorthand to roster helpers to bubble parse errors from toInt.
